    <script>
        // Constantes
        const CUSTO_HORA = 278.09; // R$ por hora

        // Estado global
        let estado = {
            tipoPrazo: null,
            unidade: 'm3',
            intervaloDias: 0,
            horasTrabalhoDia: 0,
            destinos: [],
            equipamentosDisponiveis: 0
        };

        // Função para formatar unidade
        function formatarUnidade(unidade) {
            return unidade === 'm3' ? 'm³' : 'ton';
        }

        // Função para formatar moeda
        function formatarMoeda(valor) {
            return new Intl.NumberFormat('pt-BR', {
                style: 'currency',
                currency: 'BRL'
            }).format(valor);
        }

        // Funções auxiliares
        function showSection(numero) {
            document.querySelectorAll('.section').forEach(el => el.classList.remove('active'));
            document.getElementById(`bloco${numero}`).classList.add('active');
            updateProgressBar(numero);
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function updateProgressBar(current) {
            for (let i = 0; i < 4; i++) {
                const step = document.getElementById(`step${i}`);
                if (i < current) {
                    step.classList.remove('active');
                    step.classList.add('completed');
                } else if (i === current) {
                    step.classList.remove('completed');
                    step.classList.add('active');
                } else {
                    step.classList.remove('active', 'completed');
                }
            }
        }

        function nextStep(numero) {
            if (numero === 1) {
                if (!estado.tipoPrazo) {
                    alert('Por favor, selecione o tipo de prazo');
                    return;
                }
                gerarOpcoes();
            }

            if (numero === 2) {
                if (estado.intervaloDias === 0 || estado.horasTrabalhoDia === 0) {
                    alert('Por favor, preencha todos os campos obrigatórios');
                    return;
                }
                updateResumoBloco1();
            }

            if (numero === 3) {
                if (estado.destinos.length === 0) {
                    alert('Por favor, adicione pelo menos um destino');
                    return;
                }
            }

            showSection(numero);
        }

        function previousStep(numero) {
            showSection(numero);
        }

        function selectTipoPrazo(tipo, element) {
            document.querySelectorAll('#bloco0 .option-btn').forEach(btn => btn.classList.remove('selected'));
            element.classList.add('selected');
            estado.tipoPrazo = tipo;
        }

        function selectUnidade(unidade, element) {
            document.querySelectorAll('.form-group .option-btn').forEach(btn => btn.classList.remove('selected'));
            element.classList.add('selected');
            estado.unidade = unidade;
        }

        function selectHoras(horas, element) {
            document.querySelectorAll('#bloco1 .options:last-child .option-btn').forEach(btn => btn.classList.remove('selected'));
            element.classList.add('selected');
            estado.horasTrabalhoDia = horas;
        }

        function gerarOpcoes() {
            const container = document.getElementById('intervaloOptions');
            container.innerHTML = '';

            if (estado.tipoPrazo === 'curto') {
                const opcoes = [
                    { label: 'Hoje (1 dia)', value: 1 },
                    { label: 'Alguns dias', value: 'custom' },
                    { label: 'Uma semana (7 dias)', value: 7 }
                ];

                opcoes.forEach(op => {
                    const btn = document.createElement('button');
                    btn.className = 'option-btn';
                    btn.textContent = op.label;
                    btn.onclick = () => {
                        if (op.value === 'custom') {
                            document.getElementById('diasEspecificos').focus();
                        } else {
                            document.getElementById('diasEspecificos').value = '';
                            estado.intervaloDias = op.value;
                            document.querySelectorAll('#intervaloOptions .option-btn').forEach(b => b.classList.remove('selected'));
                            btn.classList.add('selected');
                        }
                    };
                    container.appendChild(btn);
                });
            } else {
                const opcoes = [
                    { label: 'Um mês (30 dias)', value: 30 },
                    { label: 'Vários meses', value: 'custom' },
                    { label: 'Um ano (365 dias)', value: 365 }
                ];

                opcoes.forEach(op => {
                    const btn = document.createElement('button');
                    btn.className = 'option-btn';
                    btn.textContent = op.label;
                    btn.onclick = () => {
                        if (op.value === 'custom') {
                            document.getElementById('diasEspecificos').focus();
                        } else {
                            document.getElementById('diasEspecificos').value = '';
                            estado.intervaloDias = op.value;
                            document.querySelectorAll('#intervaloOptions .option-btn').forEach(b => b.classList.remove('selected'));
                            btn.classList.add('selected');
                        }
                    };
                    container.appendChild(btn);
                });
            }
        }

        document.getElementById('diasEspecificos')?.addEventListener('change', function() {
            if (this.value) {
                estado.intervaloDias = parseInt(this.value);
                document.querySelectorAll('#intervaloOptions .option-btn').forEach(b => b.classList.remove('selected'));
            }
        });

        function updateResumoBloco1() {
            const totalHoras = estado.intervaloDias * estado.horasTrabalhoDia;
            document.getElementById('resumoIntervalo').textContent = `${estado.intervaloDias} dias`;
            document.getElementById('resumoHoras').textContent = `${estado.horasTrabalhoDia}h`;
            document.getElementById('resumoTotalHoras').textContent = `${totalHoras}h`;
            document.getElementById('resumoBloco1').classList.remove('hidden');
        }

        function atualizarCamposDestino() {
            document.getElementById('producaoDestino').value = '';
            document.getElementById('tempoDestino').value = '';
            document.getElementById('capacidadeDestino').value = '';
            document.getElementById('eficienciaDestino').value = '80';
            document.getElementById('unidadeTempoDestino').value = 'min';
        }

        function adicionarDestinoSelecionado() {
            const selectElement = document.getElementById('selectDestino');
            const producao = parseFloat(document.getElementById('producaoDestino').value);
            const tempo = parseFloat(document.getElementById('tempoDestino').value);
            const unidadeTempo = document.getElementById('unidadeTempoDestino').value;
            const capacidade = parseFloat(document.getElementById('capacidadeDestino').value);
            const eficiencia = parseFloat(document.getElementById('eficienciaDestino').value) || 80;
            const destinoId = selectElement.value;

            if (!destinoId) {
                alert('Por favor, selecione um destino');
                return;
            }

            if (isNaN(producao) || producao <= 0) {
                alert('Por favor, defina uma produção válida');
                return;
            }

            if (isNaN(tempo) || tempo <= 0) {
                alert('Por favor, defina um tempo de ciclo válido');
                return;
            }

            if (isNaN(capacidade) || capacidade <= 0) {
                alert('Por favor, defina uma capacidade válida');
                return;
            }

            if (estado.destinos.some(d => d.id === destinoId)) {
                alert('Este destino já foi adicionado. Remova e adicione novamente se quiser alterar.');
                return;
            }

            const tempoMinutos = unidadeTempo === 'h' ? tempo * 60 : tempo;
            const nomeDestino = selectElement.options[selectElement.selectedIndex].text;

            const destino = {
                id: destinoId,
                nome: nomeDestino,
                producao,
                tempoMinutos,
                capacidade,
                eficiencia: Math.min(eficiencia, 100)
            };

            estado.destinos.push(destino);
            limparFormDestino();
            atualizarListaDestinos();
            document.getElementById('btnContinuarDestinos').disabled = false;
        }

        function removerDestino(id) {
            estado.destinos = estado.destinos.filter(d => d.id !== id);
            atualizarListaDestinos();
            if (estado.destinos.length === 0) {
                document.getElementById('btnContinuarDestinos').disabled = true;
            }
        }

        function atualizarListaDestinos() {
            const lista = document.getElementById('listDestinos');
            if (estado.destinos.length === 0) {
                lista.innerHTML = '<p style="color: #999;">Nenhum destino adicionado ainda</p>';
                return;
            }

            lista.innerHTML = estado.destinos.map(destino => `
                <div class="destino-card">
                    <h4>${destino.nome}</h4>
                    <p>📦 Produção: <strong>${destino.producao} ${formatarUnidade(estado.unidade)}</strong></p>
                    <p>⏱️ Ciclo: <strong>${destino.tempoMinutos.toFixed(0)} min</strong> | Capacidade: <strong>${destino.capacidade} ${formatarUnidade(estado.unidade)}</strong> | Eficiência: <strong>${destino.eficiencia}%</strong></p>
                    <button class="btn-remove" onclick="removerDestino('${destino.id}')">Remover</button>
                </div>
            `).join('');
        }

        function limparFormDestino() {
            document.getElementById('selectDestino').value = '';
            document.getElementById('producaoDestino').value = '';
            document.getElementById('tempoDestino').value = '';
            document.getElementById('capacidadeDestino').value = '';
            document.getElementById('eficienciaDestino').value = '80';
        }

        document.getElementById('equipamentosDisponiveis')?.addEventListener('change', function() {
            estado.equipamentosDisponiveis = parseInt(this.value) || 0;
        });

        function gerarRelatorio() {
            const totalHoras = estado.intervaloDias * estado.horasTrabalhoDia;
            const unidadeFormatada = formatarUnidade(estado.unidade);
            let totalEquipamentosNecessarios = 0;
            let html = `<h2>📊 Relatório de Dimensionamento de Frota</h2>`;

            // Seção 1: Configuração Geral
            html += `
                <div class="relatorio-section">
                    <h3>📅 Configuração Geral</h3>
                    <div class="relatorio-item">
                        <span>Intervalo:</span>
                        <strong>${estado.intervaloDias} dias</strong>
                    </div>
                    <div class="relatorio-item">
                        <span>Horas de trabalho por dia:</span>
                        <strong>${estado.horasTrabalhoDia}h</strong>
                    </div>
                    <div class="relatorio-item">
                        <span>Total de horas disponíveis:</span>
                        <strong>${totalHoras} horas</strong>
                    </div>
                    <div class="relatorio-item">
                        <span>Unidade:</span>
                        <strong>${unidadeFormatada}</strong>
                    </div>
                    <div class="relatorio-item">
                        <span>Custo por hora:</span>
                        <strong>${formatarMoeda(CUSTO_HORA)}</strong>
                    </div>
                </div>
            `;

            // Seção 2: Detalhes por Destino
            html += `<div class="relatorio-section"><h3>🚛 Detalhes por Destino</h3>`;

            estado.destinos.forEach((destino, index) => {
                const ciclosHora = 60 / destino.tempoMinutos;
                const producaoHora = ciclosHora * destino.capacidade * (destino.eficiencia / 100);
                const producaoNecessariaHora = destino.producao / totalHoras;
                const equipamentosNecessarios = Math.ceil(producaoNecessariaHora / producaoHora);
                const viagensNecessarias = Math.ceil(destino.producao / destino.capacidade);
                const horasTotais = destino.producao / producaoHora;
                const custoPorDestino = horasTotais * CUSTO_HORA;
                const custoPorUnidade = custoPorDestino / destino.producao;

                totalEquipamentosNecessarios += equipamentosNecessarios;

                html += `
                    <div style="background: white; padding: 15px; margin-bottom: 15px; border-radius: 8px; border-left: 4px solid #667eea;">
                        <h4>${index + 1}. ${destino.nome}</h4>
                        <div class="relatorio-item">
                            <span>Produção requerida:</span>
                            <strong>${destino.producao} ${unidadeFormatada}</strong>
                        </div>
                        <div class="relatorio-item">
                            <span>Produção necessária por hora:</span>
                            <strong>${producaoNecessariaHora.toFixed(2)} ${unidadeFormatada}/h</strong>
                        </div>
                        <div class="relatorio-item">
                            <span>Ciclos por hora:</span>
                            <strong>${ciclosHora.toFixed(2)}</strong>
                        </div>
                        <div class="relatorio-item">
                            <span>Produção por equipamento/hora:</span>
                            <strong>${producaoHora.toFixed(2)} ${unidadeFormatada}/h</strong>
                        </div>
                        <div class="relatorio-item">
                            <span>Viagens necessárias:</span>
                            <strong>${viagensNecessarias}</strong>
                        </div>
                        <div class="relatorio-item">
                            <span>Horas totais:</span>
                            <strong>${horasTotais.toFixed(2)}h</strong>
                        </div>
                        <div class="relatorio-item" style="color: #4caf50; font-weight: bold; margin-top: 10px; padding-top: 10px; border-top: 1px solid #e0e0e0;">
                            <span>Equipamentos NECESSÁRIOS:</span>
                            <strong>${equipamentosNecessarios} 🚚</strong>
                        </div>
                        <div class="relatorio-item" style="color: #2196f3; font-weight: bold; margin-top: 5px;">
                            <span>Custo da operação:</span>
                            <strong>${formatarMoeda(custoPorDestino)}</strong>
                        </div>
                        <div class="relatorio-item" style="color: #ff9800; font-weight: bold; margin-top: 5px;">
                            <span>Custo por ${unidadeFormatada}:</span>
                            <strong>${formatarMoeda(custoPorUnidade)}</strong>
                        </div>
                    </div>
                `;
            });

            html += `</div>`;

            // Seção 3: Resumo Geral
            const diferenca = totalEquipamentosNecessarios - estado.equipamentosDisponiveis;
            let classResultado = 'resultado-box';
            let mensagem = '';

            if (diferenca > 0) {
                classResultado += ' error';
                mensagem = `⚠️ FALTAM ${diferenca} equipamento(s)`;
            } else if (diferenca === 0) {
                classResultado += '';
                mensagem = `✅ PERFEITO! Você tem exatamente o que precisa!`;
            } else {
                classResultado += ' warning';
                mensagem = `✅ SOBRAM ${Math.abs(diferenca)} equipamento(s)`;
            }

            // Calcular custo total
            const custTotalOperacao = estado.destinos.reduce((total, destino) => {
                const ciclosHora = 60 / destino.tempoMinutos;
                const producaoHora = ciclosHora * destino.capacidade * (destino.eficiencia / 100);
                const horasTotais = destino.producao / producaoHora;
                return total + (horasTotais * CUSTO_HORA);
            }, 0);

            // Calcular custo por m³ (ou ton) TOTAL
            const producaoTotal = estado.destinos.reduce((total, destino) => total + destino.producao, 0);
            const custoPorUnidadeTotal = custTotalOperacao / producaoTotal;

            html += `
                <div class="relatorio-section">
                    <h3>✅ Resumo Geral</h3>
                    <div class="${classResultado}">
                        <h4>${mensagem}</h4>
                        <div class="relatorio-item">
                            <span>Total de equipamentos NECESSÁRIOS:</span>
                            <strong>${totalEquipamentosNecessarios} 🚚</strong>
                        </div>
                        <div class="relatorio-item">
                            <span>Total de equipamentos DISPONÍVEIS:</span>
                            <strong>${estado.equipamentosDisponiveis} 🚚</strong>
                        </div>
                        <div class="relatorio-item" style="margin-top: 10px; padding-top: 10px; border-top: 1px solid rgba(0,0,0,0.1);">
                            <span>Diferença:</span>
                            <strong>${Math.abs(diferenca)}</strong>
                        </div>
                    </div>

                    <div class="custo-destaque">
                        <h3>💰 Custo Total da Operação</h3>
                        <div class="custo-valor">${formatarMoeda(custTotalOperacao)}</div>
                        <p style="margin-top: 10px; opacity: 0.9;">Baseado em ${formatarMoeda(CUSTO_HORA)}/hora</p>
                    </div>

                    <div class="custo-por-unidade">
                        <h3>📊 Custo por ${unidadeFormatada}</h3>
                        <div class="custo-unidade-valor">${formatarMoeda(custoPorUnidadeTotal)}</div>
                        <p style="margin-top: 10px; opacity: 0.9;">Custo médio por unidade produzida</p>
                    </div>
                </div>
            `;

            // Seção 4: Distribuição por Destino
            html += `<div class="relatorio-section"><h3>📍 Distribuição de Equipamentos por Destino</h3>`;

            estado.destinos.forEach((destino, index) => {
                const ciclosHora = 60 / destino.tempoMinutos;
                const producaoHora = ciclosHora * destino.capacidade * (destino.eficiencia / 100);
                const producaoNecessariaHora = destino.producao / totalHoras;
                const equipamentosNecessarios = Math.ceil(producaoNecessariaHora / producaoHora);

                html += `
                    <div class="relatorio-item">
                        <span>${index + 1}. ${destino.nome}</span>
                        <strong>${equipamentosNecessarios} 🚚</strong>
                    </div>
                `;
            });

            html += `</div>`;

            document.getElementById('relatorioContent').innerHTML = html;
            showSection(4);
        }
